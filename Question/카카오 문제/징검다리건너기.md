# 징검다리 건너기

https://programmers.co.kr/learn/courses/30/lessons/64062?language=cpp

### 효율성테스트 불통과 코드
```c++
#include <string>
#include <vector>
#include <iostream>

using namespace std;

bool is_moveable(vector<int> &stones, int k, int index, int amount)
{
    int count = 0;
    
    for (int i = index; i < (int)stones.size(); i++)
    {
        if (stones[i] - amount <= 0)
            count++;
        else if (count >= k)
            break;
        else
            break;
    }
    if (count >= k)
        return false;
    return true;
}

int solution(vector<int> stones, int k)
{
    int answer = 0;
    int size = (int)stones.size();
    int minus = 0;
    int amount = 200000000;
    int last_minus = 0;
    bool is_end = false;
    
    while (!is_end)
    {
        for (int i = 0; i < size; i++)
        {
            if (!is_moveable(stones, k, i, minus + amount))
            {
                is_end = true;
                break;
            }
        }
        if (is_end && amount > 1)
        {
            amount /= 2;
            minus = last_minus;
            is_end = false;
        }
        else
        {
            answer++;
            last_minus = minus + amount;
            minus += amount;
        }
    }
    answer = minus;
    return answer;
}
```
