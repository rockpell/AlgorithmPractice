# 괄호 변환

### 문제 해결 방안
올바른 괄호 문자열을 판별하는 함수, 균형잡힌 괄호 문자열을 판별하는 함수, 문자열을 분리하는 함수를 만들어 문제의 설명대로 진행하여 해결하였습니다.

```java
class Solution {
    public String solution(String p) {
        String answer = "";

        if(isCorrect(p))
            return p;

        answer = create(p);

        return answer;
    }

    public String create(String p){
        String[] uv = separateString(p);

        if(p.length() == 0)
            return p;

        if(isCorrect(uv[0])){
            String temp = create(uv[1]);
            uv[0] += temp;
            return uv[0];
        }
        else {
            String temp = "(";
            temp += create(uv[1]);
            temp += ")";
            String temp2  = uv[0].substring(1, uv[0].length() - 1);
            if(temp2.length() > 0){
                for(int i = 0; i < temp2.length(); i++){
                    if(temp2.charAt(i) == '(')
                        temp += ')';
                    else
                        temp += '(';
                }
            }
            return temp;
        }
    }

    public String[] separateString(String text){
        String[] result = new String[2];

        for(int i = 1; i < text.length(); i++){
            String target = text.substring(0, i + 1);
            if(isBalance(target)) {
                result[0] = target;
                result[1] = text.substring(i+1);
                return result;
            }
        }
        return result;
    }

    public boolean isBalance(String text){
        int n1 = 0;
        int n2 = 0;

        for(int i = 0; i < text.length(); i++){
            if(text.charAt(i) == '(')
                ++n1;
            else
                ++n2;
        }

        if(n1 == n2)
            return true;
        else
            return false;
    }

    public boolean isCorrect(String text){
        int count = 0;

        for(int i = 0; i < text.length(); i++){
            if(text.charAt(i) == '(')
                ++count;
            else
                --count;

            if(count < 0)
                return false;
        }

        if(count > 0)
            return false;

        return true;
    }
}
```
