class Solution {
    public int[] solution(int[] heights) {       
        int[] answer = new int[heights.length];

        for(int i = 1; i < heights.length; i++){
            for(int p = i - 1; p >= 0; p--){
                if(heights[i] < heights[p]){
                    answer[i] = p + 1;
                    break;
                }
            }
        }
        
        return answer;
    }
}
